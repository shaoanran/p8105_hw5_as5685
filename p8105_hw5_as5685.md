``` r
data_function = function(filename){
    
  read_csv(file = str_c("./data/", filename))
  
}

dataframe = 
  dataframe %>% 
  mutate(data = map(dataframe$file, data_function))
```

``` r
unnest_data = 
dataframe %>% 
  unnest(data)

tidy_dataset = 
  pivot_longer(
    unnest_data,
    week_1:week_8,
```
