``` r
mean_function = function(x){
    
    if (is.numeric(x)) {
      x = replace_na(x,mean(x[!is.na(x)]))
    }else if (is.character(x)) {
      x = replace_na(x, "virginica")
    }
}
  
new_dataset = map_df( iris_with_missing, mean_function)
```
